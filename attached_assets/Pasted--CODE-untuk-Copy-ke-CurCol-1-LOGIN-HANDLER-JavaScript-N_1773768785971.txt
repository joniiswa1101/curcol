// File: services/cicoAuth.js
const CICO_API_URL = "https://workspace.joniiswa1101.repl.co";
export async function loginWithCICO(username, password) {
  try {
    const response = await fetch(`${CICO_API_URL}/api/auth/sso/login`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        username,    // username, email, atau NIK
        password
      })
    });
    const data = await response.json();
    if (!response.ok) {
      throw new Error(data.error || "Login failed");
    }
    // Simpan token di session/localStorage
    localStorage.setItem("cico_sso_token", data.token);
    localStorage.setItem("cico_user", JSON.stringify(data.user));
    return data;
  } catch (error) {
    console.error("CICO login error:", error);
    throw error;
  }
}
2. VERIFY TOKEN (Middleware)
// File: middleware/verifyCICOToken.js
const CICO_API_URL = "https://workspace.joniiswa1101.repl.co";
export async function verifyCICOToken(token) {
  try {
    const response = await fetch(`${CICO_API_URL}/api/auth/sso/validate`, {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${token}`,
        "Content-Type": "application/json"
      }
    });
    const data = await response.json();
    if (!response.ok) {
      throw new Error("Token invalid");
    }
    return data.user;  // Return user data
  } catch (error) {
    console.error("Token verification failed:", error);
    return null;
  }
}

// File: components/CICOLoginForm.jsx
import { useState } from "react";
import { loginWithCICO } from "../services/cicoAuth";
export default function CICOLoginForm() {
  const [username, setUsername] = useState("");
  const [password, setPassword] = useState("");
  const [error, setError] = useState("");
  const [loading, setLoading] = useState(false);
  async function handleLogin(e) {
    e.preventDefault();
    setError("");
    setLoading(true);
    try {
      const result = await loginWithCICO(username, password);
      console.log("Login successful:", result.user);
      
      // Redirect ke dashboard atau home page
      window.location.href = "/dashboard";
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }
  return (
    <form onSubmit={handleLogin}>
      <input
        type="text"
        placeholder="Username / Email / NIK"
        value={username}
        onChange={(e) => setUsername(e.target.value)}
        required
      />
      <input
        type="password"
        placeholder="Password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        required
      />
      {error && <div style={{ color: "red" }}>{error}</div>}
      <button type="submit" disabled={loading}>
        {loading ? "Loading..." : "Login dengan CICO"}
      </button>
    </form>
  );
}
6. PROTECTED ROUTE (Verifikasi Token)
// File: components/ProtectedRoute.jsx
import { useEffect, useState } from "react";
import { verifyCICOToken } from "../middleware/verifyCICOToken";
import { logoutCICO } from "../services/cicoAuth";
export default function ProtectedRoute({ children }) {
  const [isValid, setIsValid] = useState(null);
  useEffect(() => {
    const token = localStorage.getItem("cico_sso_token");
    if (!token) {
      window.location.href = "/login";
      return;
    }
    // Verify token
    verifyCICOToken(token).then((user) => {
      if (user) {
        setIsValid(true);
      } else {
        logoutCICO();
        window.location.href = "/login";
      }
    });
  }, []);
  if (isValid === null) {
    return <div>Loading...</div>;
  }
  return isValid ? children : null;
}
7. APP SETUP (Main Flow)
// File: App.jsx
import { useEffect, useState } from "react";
import CICOLoginForm from "./components/CICOLoginForm";
import Dashboard from "./pages/Dashboard";
import { verifyCICOToken } from "./middleware/verifyCICOToken";
export default function App() {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [loading, setLoading] = useState(true);
  useEffect(() => {
    // Check if user already logged in
    const token = localStorage.getItem("cico_sso_token");
    if (token) {
      verifyCICOToken(token).then((user) => {
        if (user) {
          setIsAuthenticated(true);
        } else {
          localStorage.removeItem("cico_sso_token");
          setIsAuthenticated(false);
        }
        setLoading(false);
      });
    } else {
      setIsAuthenticated(false);
      setLoading(false);
    }
  }, []);
  if (loading) {
    return <div>Loading...</div>;
  }
  return isAuthenticated ? <Dashboard /> : <CICOLoginForm />;
}
